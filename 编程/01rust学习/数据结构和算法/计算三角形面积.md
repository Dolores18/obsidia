```rust
use std::f64;

// 定义一个Point结构体来表示二维坐标系中的点
#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

// 实现Point结构体的方法
impl Point {
    // 创建一个新的Point
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }
}

// 计算三角形面积的函数
fn triangle_area(p1: Point, p2: Point, p3: Point) -> f64 {
    // 使用行列式计算面积
    0.5 * f64::abs(
        p1.x * (p2.y - p3.y) +
        p2.x * (p3.y - p1.y) +
        p3.x * (p1.y - p2.y)
    )
}

fn main() {
    // 创建三个点来表示三角形的顶点
    let point1 = Point::new(0.0, 0.0);
    let point2 = Point::new(4.0, 0.0);
    let point3 = Point::new(2.0, 3.0);

    // 计算三角形的面积
    let area = triangle_area(point1, point2, point3);

    // 打印结果
    println!("三角形的面积是: {:.2}", area);
}
```

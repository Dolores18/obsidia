use std::fmt::Display;

struct Point<T> {
    x: T,
    y: T,
}

impl<T> Point<T> 
where 
    T: Display + Copy + From<i32>,
{
    fn new(x: T, y: T) -> Self {
        Point { x, y }
    }

    fn origin() -> Self {
        Point { x: T::from(0), y: T::from(0) }
    }

    fn display(&self) {
        println!("Point at ({}, {})", self.x, self.y);
    }
}

fn main() {
    let p1: Point<i32> = Point::new(5, 10);
    p1.display();

    let p2: Point<f64> = Point::new(1.0, 2.0);
    p2.display();

    let p3: Point<i32> = Point::origin();
    p3.display();

    // 这行会编译错误，因为 String 不满足约束条件
    // let p4: Point<String> = Point::new(String::from("a"), String::from("b"));
}

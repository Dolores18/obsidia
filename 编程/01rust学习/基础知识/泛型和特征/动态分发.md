
```rust
trait Animal {
    fn make_sound(&self);
}

struct Dog;
impl Animal for Dog {
    fn make_sound(&self) {
        println!("Woof!");
    }
}

struct Cat;
impl Animal for Cat {
    fn make_sound(&self) {
        println!("Meow!");
    }
}

fn animal_sound(animal: Box<dyn Animal>) {
    animal.make_sound();
}

fn main() {
    let dog = Box::new(Dog);
    let cat = Box::new(Cat);

    animal_sound(dog); // 输出: Woof!
    animal_sound(cat); // 输出: Meow!
}
```

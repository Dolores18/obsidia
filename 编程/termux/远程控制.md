# 更新软件包
pkg upgrade
# 安装OpenSSH
pkg install termux-auth openssh
# 修改用户密码
passwd   和手机锁屏一样
# 开启SSH服务
sshd
# 获取用户名    
whoami    u0_a481
# 获取IP
ifconfig    192.168.101.35
# SSH连接
ssh -p 8022 user@host
# 关闭SSH服务
pkill sshd

# 永久启用ssh服务
# 在F-Droid安装Termux:Boot APP
pkg install termux-services
sv-enable sshd
cat > ~/.termux/boot/boot.sh<< EOF
termux-wake-lock  
sv up sshd
EOF
